-- 1. Create the 'devices' table
CREATE TABLE IF NOT EXISTS devices (
  -- The 8-digit code acts as the unique identifier (Primary Key)
  code TEXT PRIMARY KEY,
  
  -- The local IP address of the ESP32 (e.g., 192.168.1.15)
  local_ip TEXT NOT NULL,
  
  -- Automatically track when the device last checked in
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 2. Enable Row Level Security (RLS)
-- This is a Supabase requirement for security
ALTER TABLE devices ENABLE ROW LEVEL SECURITY;

-- 3. Create a Public Access Policy
-- This allows your Serverless Functions to Insert, Update, and Read data
-- without needing complex Auth logic for this specific project.
CREATE POLICY "Enable public access for all operations" ON devices
FOR ALL
USING (true)
WITH CHECK (true);

-- 4. (Optional) Create a function to automatically update 'updated_at' on every change
CREATE OR REPLACE FUNCTION update_modified_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = now();
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_devices_modtime
    BEFORE UPDATE ON devices
    FOR EACH ROW
    EXECUTE PROCEDURE update_modified_column();
