            <button class="btn-primary" onclick="connect()">CONNECT SYSTEM</button>

            <h2 id="deviceStatus">Connected</h2>

            <button style="background: #10b981; color: white;" onclick="hardwareCmd('/led/on')">LED ON</button>
            <button style="background: #ef4444; color: white;" onclick="hardwareCmd('/led/off')">LED OFF</button>
            
            <div style="margin-top: 30px; border-top: 1px solid #334155; padding-top: 20px;">

                <button class="btn-danger" onclick="triggerWiFi()">RESET WIFI SETTINGS</button>

    <script>
        let espIP = "";

        async function connect() {
            const code = document.getElementById('code').value;
            if(code.length !== 8) return alert("Code must be 8 digits");
            
            try {
                // 1. Lookup IP from Vercel API
                const res = await fetch(`/api/lookup?code=${code}`);
                if(!res.ok) throw new Error();
                const data = await res.json();
                
                espIP = data.ip;
                
                // 2. Double check connection to the ESP32 directly
                // Note: We use check_pin to ensure the IP matches the Code
                const test = await fetch(`http://${espIP}/check_pin?pin=${code}`);
                if(test.ok) {
                    document.getElementById('authView').classList.add('hidden');
                    document.getElementById('controlView').classList.remove('hidden');
                    document.getElementById('ipLabel').innerText = `Target IP: ${espIP}`;
                } else {
                    alert("IP found, but hardware rejected the code.");
                }
            } catch (e) { 
                alert("Device not found. Check Serial Monitor for code."); 
            }
        }

        function hardwareCmd(path) {
            // mode: 'no-cors' allows browser to ignore the HTTPS/HTTP mismatch
            fetch(`http://${espIP}${path}`, { mode: 'no-cors' })
                .catch(() => console.log("Packet sent (ignoring CORS response)"));
        }

        async function triggerWiFi() {
            if(confirm("CRITICAL: This will wipe WiFi settings. You will need to reconnect to 'ESP32_Nexus_Setup'. Continue?")) {
                hardwareCmd('/reset_wifi');
                alert("Reset command sent. Hardware is rebooting. Please check your phone's WiFi settings in 10 seconds.");
                location.reload();
            }
        }
    </script>
