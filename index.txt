<script>
    let state = { game: '', players: 4, isLocked: true, min: 1, max: 10 };

    function checkPin() {
        if(document.getElementById('pinCode').value === "12345678") {
            document.getElementById('authView').classList.add('hidden');
            document.getElementById('controlView').classList.remove('hidden');
        } else { alert("Access Denied"); }
    }

    function setGame(name, count, adjustable, min=1, max=10) {
        state.game = name; state.players = count; state.isLocked = !adjustable;
        state.min = min; state.max = max;
        document.querySelectorAll('.game-card').forEach(c => c.classList.toggle('active', c.innerText === name));
        document.getElementById('playerVal').innerText = count;
        
        const m = document.getElementById('minus'), p = document.getElementById('plus');
        m.classList.toggle('disabled', !adjustable); m.innerHTML = adjustable ? '−' : '🔒';
        p.classList.toggle('disabled', !adjustable); p.innerHTML = adjustable ? '+' : '🔒';
    }

    document.getElementById('plus').onclick = () => { if(!state.isLocked && state.players < state.max) { state.players++; document.getElementById('playerVal').innerText = state.players; } };
    document.getElementById('minus').onclick = () => { if(!state.isLocked && state.players > state.min) { state.players--; document.getElementById('playerVal').innerText = state.players; } };

    function submitSetup() {
        if(!state.game) return alert("Select a game");
        if(state.game === 'Omi') {
            document.getElementById('selectionArea').classList.add('hidden');
            document.getElementById('omiTrumpView').classList.remove('hidden');
        } else {
            showFinal(`Game: ${state.game}<br>No. of Players: ${state.players}`);
        }
    }

    function finishOmi(trump) {
        document.getElementById('omiTrumpView').classList.add('hidden');
        showFinal(`Game: Omi<br>No. of Players: 4<br>Trump Card: ${trump}`);
    }

    function showFinal(html) {
        document.getElementById('selectionArea').classList.add('hidden');
        document.getElementById('resultView').classList.remove('hidden');
        document.getElementById('finalText').innerHTML = `<strong>SESSION ACTIVE</strong><br>${html}`;
    }

    // Clears the current game selection but stays logged in
    function resetUI() {
        state = { game: '', players: 4, isLocked: true, min: 1, max: 10 };
        document.querySelectorAll('.game-card').forEach(c => c.classList.remove('active'));
        document.getElementById('playerVal').innerText = '4';
        document.getElementById('selectionArea').classList.remove('hidden');
        document.getElementById('omiTrumpView').classList.add('hidden');
        document.getElementById('resultView').classList.add('hidden');
    }

    // Wipes everything and goes back to PIN entry
    function disconnect() {
        resetUI();
        document.getElementById('pinCode').value = '';
        document.getElementById('controlView').classList.add('hidden');
        document.getElementById('authView').classList.remove('hidden');
    }
</script>
