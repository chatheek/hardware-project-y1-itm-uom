#include <WiFi.h>
#include <WebServer.h>
#include <HTTPClient.h>
#include <WiFiManager.h>
#include <Wire.h>
#include <U8g2lib.h>

#define STEP_PIN 12
#define DIR_PIN 14
#define LED_PIN 2

// 1.3" SH1106 OLED
U8G2_SH1106_128X64_NONAME_F_HW_I2C u8g2(U8G2_R0, U8X8_PIN_NONE);

WebServer server(80);
String deviceCode = "";
const char* registerUrl = "https://hardware-project-y1-itm-uom.vercel.app/api/register";

// Corrected Gear Math: (102 teeth / 18 teeth) * 200 steps = 1133.33
const float stepsPerPlateRotation = 1133.33; 

void displayGameStatus(String game, int players, String trump = "") {
    u8g2.clearBuffer();
    u8g2.setFont(u8g2_font_helvB14_tr);
    u8g2.drawStr(0, 20, game.c_str());
    
    u8g2.setFont(u8g2_font_9x15_tr);
    String pLine = "Players: " + String(players);
    u8g2.drawStr(0, 42, pLine.c_str());

    if(trump != "") {
        u8g2.setFont(u8g2_font_unifont_t_symbols); // Supports suit symbols
        String tLine = "Trump: " + trump;
        u8g2.drawStr(0, 62, tLine.c_str());
    }
    u8g2.sendBuffer();
}

void rotateToPlayers(int players) {
    if (players <= 1) return; // Skip rotation for 1 player (Solitaire)
    
    int stepsPerStop = (int)(stepsPerPlateRotation / players);
    digitalWrite(DIR_PIN, HIGH);
    
    for(int i = 0; i < players; i++) {
        for(int x = 0; x < stepsPerStop; x++) {
            digitalWrite(STEP_PIN, HIGH);
            delayMicroseconds(1200);
            digitalWrite(STEP_PIN, LOW);
            delayMicroseconds(1200);
        }
        delay(1500); // Pause at each player
    }
}

void setup() {
    Serial.begin(115200);
    pinMode(STEP_PIN, OUTPUT); pinMode(DIR_PIN, OUTPUT); pinMode(LED_PIN, OUTPUT);
    
    u8g2.begin();
    WiFiManager wm;
    wm.autoConnect("ESP32_Nexus_Setup");

    randomSeed(analogRead(0));
    for(int i=0; i<8; i++) deviceCode += String(random(0,10));

    // Display PIN for Login
    u8g2.clearBuffer();
    u8g2.setFont(u8g2_font_logisoso24_tr);
    u8g2.drawStr(10, 50, deviceCode.c_str());
    u8g2.sendBuffer();

    HTTPClient http;
    http.begin(registerUrl);
    http.addHeader("Content-Type", "application/json");
    String json = "{\"code\":\"" + deviceCode + "\", \"ip\":\"" + WiFi.localIP().toString() + "\"}";
    http.POST(json);
    http.end();

    server.enableCORS();
    
    server.on("/start_game", []() {
        String game = server.arg("game");
        int players = server.arg("players").toInt();
        String trump = server.arg("trump");
        server.send(200);

        displayGameStatus(game, players, trump);
        rotateToPlayers(players);
    });

    server.on("/check_pin", []() {
        if (server.arg("pin") == deviceCode) server.send(200);
        else server.send(403);
    });

    server.begin();
}

void loop() { server.handleClient(); }
